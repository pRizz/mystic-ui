## Usage

```tsx
import { AuroraText } from "@/components/mystic-ui/aurora-text";

export default function Example() {
	return <AuroraText>Aurora Text</AuroraText>;
}
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `colors` | `string[]` | `["#FF0080", "#7928CA", "#0070F3", "#38bdf8"]` | Gradient color stops used for the aurora effect. |
| `speed` | `number` | `1` | Speed multiplier for the aurora animation. |
